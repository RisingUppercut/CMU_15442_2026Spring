# Load: all threads copy global → shared

    Tx.copy(Asmem[:,:], A[m_st:m_st+128, k_st:k_st+64])

Tx.cuda.cta_sync()
# Compute: single thread issues MMA

Tx.ptx.mbarrier.try_wait(mma_bar, phase)
